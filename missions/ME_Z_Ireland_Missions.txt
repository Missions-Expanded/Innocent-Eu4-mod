ireland_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = IRE
		OR = {
			has_country_flag = use_missions_expanded_missions_instead_of_paradox
			NOT = { has_dlc = "Rule Britannia" }
		}
	}
	has_country_shield = yes

	ire_gaelic_irish = {
		icon = mission_alliances
		required_missions = { irish_minor_eirigh_dhuiche_dia }
		position = 7
		trigger = {
			num_of_owned_provinces_with = {
				value = 5
				development = 15
				culture = irish
			}
		}
		effect = {
			add_country_modifier = {
				name = ire_rahunas_na_heireann
				duration = 7300
			}
		}
	}

	ire_trinity_college = {
		icon = mission_alliances
		required_missions = { ire_gaelic_irish }
		position = 8
		provinces_to_highlight = {
			province_id = 373
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { development = 25 }
				devastation = 0
				NOT = { has_tax_building_trigger = yes }
				NOT = { has_production_building_trigger = yes }
			}
		}
		trigger = {
			373 = {
				owned_by = ROOT
				devastation = 0
				development = 25
				has_tax_building_trigger = yes
				has_production_building_trigger = yes
			}
		}
		effect = {
			373 = {
				add_building = university
				add_permanent_province_modifier = {
					name = ire_trinity_college
					duration = -1
				}
			}
		}
	}

	ire_public_culture = {
		icon = mission_alliances
		required_missions = {
			ire_trinity_college
			ire_economic_attempts
		}
		position = 10
		provinces_to_highlight = {
			OR = {
				area = ulster_area
				area = munster_area
				area = connacht_area
				area = leinster_area
			}
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { has_building = university }
			}
		}
		trigger = {
			ulster_area = {
				type = all
				owned_by = ROOT
				has_building = university
			}
			munster_area = {
				type = all
				owned_by = ROOT
				has_building = university
			}
			connacht_area = {
				type = all
				owned_by = ROOT
				has_building = university
			}
			leinster_area = {
				type = all
				owned_by = ROOT
				has_building = university
			}
		}
		effect = {
			add_splendor = 100
			add_mana_scaled_on_monarch_skill = yes
		}
	}

	ire_celtic_union = {
		icon = mission_alliances
		required_missions = { ire_public_culture }
		position = 11
		trigger = {
			development_in_provinces = {
				value = 250
				OR = {
					area = ulster_area
					area = munster_area
					area = connacht_area
					area = leinster_area
				}
			}
		}
		effect = {
			ulster_area = {
				add_permanent_province_modifier = {
					name = ire_celtic_union_modifier
					duration = -1
				}
			}
			munster_area = {
				add_permanent_province_modifier = {
					name = ire_celtic_union_modifier
					duration = -1
				}
			}
			connacht_area = {
				add_permanent_province_modifier = {
					name = ire_celtic_union_modifier
					duration = -1
				}
			}
			leinster_area = {
				add_permanent_province_modifier = {
					name = ire_celtic_union_modifier
					duration = -1
				}
			}
		}
	}

	ire_high_kingship = {
		icon = mission_alliances
		required_missions = { ire_celtic_union }
		position = 12
		provinces_to_highlight = {
			OR = {
				area = ulster_area
				area = munster_area
				area = connacht_area
				area = leinster_area
			}
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { development = 30 }
			}
		}
		trigger = {
			ulster_area = {
				type = all
				owned_by = ROOT
				development = 30
				if = {
					limit = { has_dlc = "Mandate of Heaven" }
					prosperity = 50
				}
			}
			munster_area = {
				type = all
				owned_by = ROOT
				development = 30
				if = {
					limit = { has_dlc = "Mandate of Heaven" }
					prosperity = 50
				}
			}
			connacht_area = {
				type = all
				owned_by = ROOT
				development = 30
				if = {
					limit = { has_dlc = "Mandate of Heaven" }
					prosperity = 50
				}
			}
			leinster_area = {
				type = all
				owned_by = ROOT
				development = 30
				if = {
					limit = { has_dlc = "Mandate of Heaven" }
					prosperity = 50
				}
			}
		}
		effect = {
			add_mana_scaled_on_monarch_skill = yes
			add_prestige = 20
			add_legitimacy_equivalent = { value = 30 }
			add_power_projection = {
				type = mission_rewards_power_projection
				amount = 20
			}
		}
	}
