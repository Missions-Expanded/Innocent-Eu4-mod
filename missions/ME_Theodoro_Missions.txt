	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = FEO
	}
	has_country_shield = yes

	theodoro_mission_1 = {
		icon = mission_alliances
		required_missions = { }
		position = 1
		trigger = {
			if = {
				limit = {
					exists = TRE
				}
				alliance_with = TRE
			}
			else = {
				num_of_allies = 1
			}
		}
		effect = {
			add_country_modifier = {
				name = "theo_alliance"
				duration = 9125
			}
		}
	}

	theodoro_mission_5 = {
		icon = mission_have_two_subjects
		required_missions = { theodoro_mission_1 }
		position = 3
		trigger = {
			total_development = 50
			if = {
				limit = {
					exists = TRE
				}
				reverse_has_opinion = {
					who = TRE
					value = 150
				}
			}
			else = {
				any_ally = {
					has_opinion = {
						who = ROOT
						value = 150
					}
				}
			}
		}
		effect = {
			add_dip_power = 100
		}
	}
}
