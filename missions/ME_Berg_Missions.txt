BRG_Missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
		OR = {
			tag = BRG
			AND = {
				tag = RHN
				was_tag = BRG
			}
		}
	}
	has_country_shield = yes
	
	BRG_succession_of_kleves = {
		icon = spread_the_von_wettins
		required_missions = { rhn_generic_fc }
		position = 2
		provinces_to_highlight = {
			OR = {
				province_id = 2975
				province_id = 84
				province_id = 4768
			}
			NOT = { owned_by = ROOT }
		}
		trigger = {
			2975 = {
				owned_by = ROOT
			}
			84 = {
				owned_by = ROOT
			}
			4768 = {
				owned_by = ROOT
			}
		}
		effect = {
			custom_tooltip = BRG_gain_claims_on_cleves
			hidden_effect = {
				KLE = {
					every_owned_province = {
						add_permanent_claim = ROOT 
					}
				}
			}
			add_country_modifier = {
				name = BRG_julich_kleves
				duration = 9125
			}
			ME_override_country_name = { name = BRGKLE }
		}
	}
	
	BRG_gelre = {
		icon = adjust_the_general_means
		required_missions = { rhn_generic_expand }
		position = 4
		provinces_to_highlight = {
			owned_by = GEL
		}
		trigger = {
			OR = {
				has_won_war_against = {
					who = GEL
					max_years_since = 100
				}
				AND = {
					NOT = { exists = GEL }
					owns = 99
				}
			}
		}
		effect = {
			add_army_tradition_or_mil_power = { amount = 15 }
			add_power_projection = {
				type = mission_rewards_power_projection
				amount = 20
			}
			add_country_modifier = {
				name = BRG_order_of_the_horn
				duration = 10950
			}
		}
	}
}
