Magdeburg_slot_2 = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		tag = MAG	
	}
	has_country_shield = yes
	
	MAG_conquest_of_anhalt = {
		icon = mission_conquer_50_development
		required_missions = {
			MAG_rally_the_men 
			MAG_find_alliances
		}
		position = 2
		provinces_to_highlight = {
			province_id = 58
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { is_core = ROOT }
			}
		}	
		trigger = {
			owns_core_province = 58
		}
		effect = {
			SAX = {
				every_owned_province = {
					limit = {
						any_neighbor_province = {
							owned_by = ROOT
						}
					}
					add_claim = ROOT
				}
			}	
		}
	}
	
	MAG_saxon_frontier = {
		icon = mission_imperial
		required_missions = { MAG_conquest_of_anhalt }
		position = 3
		provinces_to_highlight = {
			is_core = SAX
			NOT = { owned_by = ROOT }
		}	
		trigger = {
			all_province = {
				OR = {
					NOT = { is_core = SAX }
					owned_by = ROOT
				}
			}
			NOT = { exists = SAX }
		}
		effect = {
			if = {
				limit = {
					NOT = { num_of_electors = 7 }
				}
				elector = yes
			}
			else_if = {
				limit = {
					50 = {
						NOT = { owned_by = ROOT }
						NOT = { is_core = ROOT }
					}
				}
				add_claim = 50
			}
			else = {
				add_prestige = 10
			}
		}
	}
}
