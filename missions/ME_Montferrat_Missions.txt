montferrat_missions_slot_3 = {
	slot = 3
	generic = no
	ai = yes
	potential = {
		tag = MFA
	}
	has_country_shield = yes
	
	mfa_resurgence = {
		icon = me_mission_italy_army1
		position = 1
		required_missions = { }
		trigger = {
			if = {
				limit = {
					is_subject = yes
				}
				num_of_generals = 1
				army_size_percentage = 1
				manpower_size_percentage = 0.33
				OR = {
					if = {
						limit = {
							exists = FRA
						}
						FRA = {
							has_opinion = {
								who = ROOT
								value = 100
							}
						}
					}
					if = {
						limit = {
							exists = MLO
						}
						MLO = {
							has_opinion = {
								who = ROOT
								value = 100
							}
						}
					}
					any_country = {
						is_rival = SAV
						OR = {
							NOT = { tag = FRA }
							NOT = { tag = MLO }
						}
						has_opinion = {
							who = ROOT
							value = 125
						}
					}
				}
			}
			else = {
				stability = 1
			}
		}
		effect = {
			if = {
				limit = {
					is_subject = yes
				}
				add_country_modifier = {
					name = mfa_for_victory_or_death
					duration = 7300
				}
				country_event = {
					id = ME_Montferrat_Events.1
				}
			}
			else = {
			}
		}
	}
	
	mfa_rising_power = {
