ME_incident_Athens_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = ATH 
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
		has_country_flag = ME_TUR_Change_Tree
	}
	has_country_shield = yes

	ATH_incident_religion_water = {
		icon = mission_hands_praying
		required_missions = { ATH_incident_first_sighting }
		position = 3
		trigger = {
			hidden_trigger = {
				TUR = {
					is_in_war = {
						casus_belli = ME_TUR_navy
					}
				}
			}
			if = {
				limit = {
					exists = PAP
					PAP = {
						is_in_war = {
							attacker_leader = TUR
							casus_belli = ME_TUR_navy
						}
						navy_size = 5
					}
				}
				current_war_losses = {
					who_with = TUR
                	type = ships
                    cause = battle
                    amount = 15
				}
			}
			else = { always = yes }
		}
		effect = {
			if = {
				limit = { 
					TUR = {
						is_crusade_target = yes 
					}
				}
				TUR = { set_country_flag = ME_TUR_crusade_stop }
			}
			else = {
				TUR = {
					add_country_modifier = {
						name = ME_TUR_severed_chrisitan_head
						duration = 1825
					}
				}
			}
		}
	}
}
