WBG_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
		OR = {
			tag = WBG
			AND = {
				was_tag = WBG
				tag = FKN
			}
		}
	}
	has_country_shield = yes
	
	WBG_papal = {
		icon = mission_religious
		trigger = {
			if = {
				limit = {
					exists = PAP
				}
				reverse_has_opinion = {
					who = PAP
					value = 100
				}
			}
			else = {
				papal_influence = 50
			}
		}
		effect = {
			add_papal_influence = 15
			add_dip_power = 50
		}
	}
	
	WBG_bishoprics = {
		icon = catholic_confraternity
		required_missions = { WBG_papal }
		position = 3
		trigger = {
			if = {
				limit = {
					ai = yes
				}
				any_ally = {
					government = theocracy
				}
			}
			else = {
				calc_true_if = {
					all_ally = {
						government = theocracy
					}
					amount = 2
				}
			}
		}
		effect = {
			add_dip_power = 50
			add_country_modifier = {
				name = WBG_diplomacy
				duration = 10950
			}
		}
	}
	
	WBG_wurzburg_residenz = {
		icon = take_the_bishoprics
		required_missions = { WBG_bishoprics }
		position = 5
		provinces_to_highlight = {
			province_id = 79
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { development = 30 }
			}
		}
		trigger = {
			if = {
				limit = {
					ai = yes
				}
				79 = {
					owned_by = ROOT
					development = 25
				}
			}
			else = {
				79 = {
					owned_by = ROOT
					development = 30
				}
			}
		}
		effect = {
			79 = {
				add_province_modifier = {
					name = WBG_wurzburg_residenz_modifier
					duration = -1
				}
			}
		}
	}
}
