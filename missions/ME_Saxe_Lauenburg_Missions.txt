Saxe_Lauenburg_missions_5 = {
	slot = 5
	ai = yes
	generic = no
	potential = {
		tag = LAU
	}
	has_country_shield = yes
	
	LAU_church = {
		icon = mission_alliances
		required_missions = { }
		position = 1
		trigger = {
			adm_power = 50
			treasury = 100
		}
		effect = {
			1857 = {
				if = {
					limit = {
						owned_by = ROOT
						NOT = { has_building = temple }
						NOT = { has_building = cathedral }
					}
					add_building = temple
				}
			}
			add_adm_power = -25
			add_treasury = -50
		}
	}
	
	LAU_trade_route = {
		icon = mission_alliances
		required_missions = { LAU_church }
		position = 2
		provinces_to_highlight = {
			province_id = 1857
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { has_building = marketplace }
			}
		}
		trigger = {
			owns = 1857
			1857 = {
				has_building = marketplace
			}
		}
		effect = {
			add_country_modifier = {
				name = HRE_booming_trade
				duration = 9125
			}
		}
	}
	
	LAU_reformation = {
		icon = mission_alliances
		required_missions = { LAU_trade_route }
		position = 3
		trigger = {
			OR = {
				is_defender_of_faith = yes
				religion = protestant
				religion = reformed
			}
		}
		effect = {
			1857 = {
				if = {
					limit = {
						owned_by = ROOT
					}
					if = {
						limit = {
							NOT = { has_building = cathedral }
						}
						add_building = cathedral
					}
					else = {
						add_base_tax = 2
					}
				}
				else = {
					add_permanent_claim = ROOT
				}
			}
		}
	}
	
	LAU_capital = {
		icon = mission_alliances
		required_missions = { LAU_reformation }
		position = 4
		provinces_to_highlight = {
