NUM_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		has_country_flag = use_missions_expanded_missions_instead_of_paradox
		OR = {
			tag = NUM
			AND = {
				was_tag = NUM
				tag = FKN
			}
		}
	}
	has_country_shield = yes
	
	NUM_city_above_all = {
		icon = ansbach_succession
		provinces_to_highlight = {
			province_id = 67
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { higher_development_than = 134 }
				NOT = { higher_development_than = 266 }
				NOT = { development = 25 }
			}
		}
		trigger = {
			67 = {
				owned_by = ROOT
				higher_development_than = 134
				higher_development_than = 266
				development = 25
			}
		}
		effect = {
			67 = {
				add_province_modifier = {
					name = NUM_Hauptstadt
					duration = -1
				}
			}
		}
	}
	
	NUM_german_renaissance = {
		icon = reclaim_the_ducal_lands
		required_missions = { NUM_city_above_all }
		position = 3
		trigger = {
			has_institution = renaissance
		}
		effect = {
			if = {
				limit = {
					current_age = age_of_discovery
				}
				add_country_modifier = {
					name = NUM_renaissance_modifier
					desc = until_age_of_discovery_ends
					duration = -1
				}
			}
			else = {
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
		}
	}
	
	NUM_stadtbiblothek = {
		icon = me_mission_man_in_library
		position = 5
		required_missions = { NUM_german_renaissance }
		provinces_to_highlight = {
			province_id = 67
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { num_of_times_improved = 10 }
			}
		}
		trigger = {
			67 = {
				owned_by = ROOT
				num_of_times_improved = 10
			}
		}
		effect = {
			67 = {
				add_institution_embracement = {
					which = printing_press
					value = 50
				}
				add_province_modifier = {
					name = NUM_stadtbibliothek_modifier
					duration = -1
				}
			}
		}
	}
