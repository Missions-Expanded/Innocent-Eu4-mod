		}

				limit = { has_estate = estate_church }
				NOT = {
					estate_influence = {

					}
				}
			}
			else = {
				absolutism = 20
			}
			num_of_owned_provinces_with = {
				value = 20
				has_state_patriach = yes
			}
			all_owned_province = {  
				OR = {
					NOT = { local_autonomy = 20 }
					NOT = { has_tax_building_trigger = yes }
				}
			} 
		}
		effect = {
			if = {
				limit = { has_estate = estate_church }
				change_estate_land_share = {
					estate = estate_church
					share = -5
				}
			}
			else = {
				add_absolutism = 10
				add_years_of_income = 0.5
			}
			if = {
				limit = { has_country_modifier = rus_udelnye_krestiane_modifier }
				add_adm_power = 250
			}
			else_if = {
				limit = { has_country_modifier = rus_podnatnaya_reforma_modifier }
				add_adm_power = 200
			}
			else_if = {
				limit = { has_country_modifier = rus_sobornoye_ulozheniye_modifier }
				add_adm_power = 150
			}
			else_if = {
				limit = { has_country_modifier = rus_russian_serfdom_modifier }
				add_adm_power = 100
			}
			else = {
				add_adm_power = 50
			}
		}
	}
