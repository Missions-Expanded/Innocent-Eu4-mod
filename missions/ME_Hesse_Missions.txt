	HES_mainz_feud = {
		icon = expand_the_confederacy
		required_missions = { rhn_generic_fc }
		position = 2
		provinces_to_highlight = {
			if = {
				limit = {
					exists = MAI
				}
				owned_by = MAI
			}
			else = {
				province_id = 78
				NOT = { owned_by = ROOT }
			}
		}
		trigger = {
			OR = {
				has_won_war_against = {
					who = MAI
				}
				AND = {
					NOT = { exists = MAI }
					owns = 78
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = HES_peace_of_frankfurt
				duration = 7300
			}
			thuringia_area = {
				limit = {
					NOT = { owned_by = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
			if = {
				limit = {
					exists = THU
					valid_for_personal_unions_trigger = yes
				}
				custom_tooltip = HES_thu_valid_yes
				hidden_effect = {
					add_casus_belli = {
						target = THU
						type = cb_restore_personal_union
						months = -1
					}
				}
			}
			else = {
				custom_tooltip = HES_thu_valid_no
			}
			tooltip = {
				add_casus_belli = {
					target = THU
					type = cb_restore_personal_union
					months = -1
				}
			}
		}
	}

	HES_thuringia = {
		icon = become_king

		position = 4
		provinces_to_highlight = {
