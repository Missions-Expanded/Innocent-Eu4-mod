evenk_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			is_evenki_minor = yes
			AND = {
				was_evenki_minor = yes
				OR = {
					tag = MCH
					tag = QNG
				}
				has_country_flag = use_missions_expanded_missions_instead_of_paradox
			}
		}
	}
	has_country_shield = yes
	
	evenk_gather_herd = {
		icon = mission_non-western_cavalry_raid
		required_missions = { evenk_adopt_firearms }
		position = 2
		provinces_to_highlight = {
			is_capital_of = ROOT
			NOT = { base_manpower = 5 }
		}
		trigger = {
			army_tradition = 20
			num_of_cavalry = 4
			capital_scope = {
				base_manpower = 5
			}
		}
		effect = {
			if = {
				limit = {
					is_expanded_mod_active = { mod = governments }
				}
				add_country_modifier = {
					name = evenk_reindeer_cavalry_ge
					duration = -1
				}
			}
			else = {
				add_country_modifier = {
					name = evenk_reindeer_cavalry
					duration = -1
				}
			}
		}
	}
	
	evenk_siberian_land_survey = {
		icon = mission_scandinavian_port_authority

		}

				id = ME_Evenk_Events.1
			}
		}
	}
}
