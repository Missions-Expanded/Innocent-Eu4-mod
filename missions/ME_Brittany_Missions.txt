	generic = no
	ai = yes
	potential = {
		tag = BRI
	}
	has_country_shield = yes
	
	BRI_side_with_england = {
		icon = mission_alliances
		position = 1
		required_missions = { }
		trigger = {
			alliance_with = ENG
			custom_trigger_tooltip = {
				tooltip = NOT_FLAG_FRA
				NOT = {
					has_country_flag = brittany_france
				}
			}
		}
		effect = {		
			set_country_flag = brittany_england
			ENG = {
				create_guarantee = BRI
			}	
			swap_non_generic_missions = yes
		}
	}
}

brittany_missions_england = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		tag = BRI
		has_country_flag = brittany_england
	}
	has_country_shield = yes
	
	BRI_make_england_likey = {
		icon = mission_noble_council
		position = 2
		required_missions = { BRI_side_with_england }
		trigger = {
			if = {
				limit = { exists = ENG }
				NOT = { is_rival = ENG }
				ENG = {
					NOT = { is_rival = ROOT }
					has_opinion = {
						who = ROOT
						value = 175
					}
				}
			}
			alliance_with = ENG
		}
		effect = {		
			if = {
				limit = { 
					has_dlc = "The Cossacks"
				}	
				add_trust = {
					who = ENG
					value = 10
					mutual = yes
				}	
			}
			else = {
				add_dip_power = 125
			}	
		}
	}
	
	BRI_marry_england = {
		icon = mission_monarch_in_throne_room
		position = 3
		required_missions = { BRI_make_england_likey }
		trigger = {
			marriage_with = ENG
			alliance_with = ENG
		}
		effect = {		
			add_country_modifier = {
				name = BRI_royal_marriages
				duration = 7300
			}
		}
	}
	
	BRI_call_england_into_war = {
		icon = mission_build_up_to_force_limit
		position = 4
		required_missions = { BRI_marry_england }
		trigger = {
			any_neighbor_country = {
				is_in_war = {
					attackers = ROOT
					attackers = ENG
				}
			}
			alliance_with = ENG				
		}
		effect = {		
			random_neighbor_country = {
				limit = {
					war_with = ROOT
				}
				random_owned_province = {
					limit = {
						any_neighbor_province = {
							owned_by = ROOT
						}
					}
					cede_province = ROOT
				}
			}
		}
	}
	
	BRI_use_england_for_france = {
		icon = mission_conquer_50_development
		position = 5
		required_missions = { BRI_call_england_into_war }
		trigger = {
			FRA = {
				is_in_war = {
					attackers = ROOT
					attackers = ENG
				}
			}			
		}
		effect = {		
			FRA = {
				if = {
					limit = {
						war_with = ROOT
						any_owned_province = {
							any_neighbor_province = {
								owned_by = ROOT
							}
						}
					}
					random_owned_province = {
						limit = {
							any_neighbor_province = {
								owned_by = ROOT
							}
						}
						cede_province = ROOT
					}
				}
			}
			FRA = {
				if = {
					limit = {
						war_with = ROOT
						war_with = ENG
						any_owned_province = {
							any_neighbor_province = {
								owned_by = ENG
							}
						}
					}
					random_owned_province = {
						limit = {
							any_neighbor_province = {
								owned_by = ENG
							}
						}
						cede_province = ENG
					}
				}
			}
		}
	}
}
