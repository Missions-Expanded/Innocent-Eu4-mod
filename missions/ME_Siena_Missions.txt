siena_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = SIE
			AND = {
				was_tag = SIE
				OR = {
					tag = TUS
					AND = {
						tag = ITA
						NOT = { was_tag = SPI }
						NOT = { was_tag = TTS }
						country_was_not_more_than_one_italian_formable = yes
					}
				}
			}
		}
		OR = {
			NOT = { has_dlc = "Emperor" }
			AND = {
				has_dlc = "Emperor"
				has_country_flag = use_missions_expanded_missions_instead_of_paradox
			}
		}
	}
	has_country_shield = yes

	sie_old_rivalries = {
		icon = florentine_renaissance
		required_missions = { }
		position = 1
		trigger = {
			if = {
				limit = { exists = LAN }
				OR = {
					alliance_with = LAN
					is_in_trade_league_with = LAN
					LAN = {
						has_opinion = {
							who = ROOT 
							value = 125
						}
					}
					is_rival = LAN
					AND = {
						is_enemy = LAN
						any_country = {
							is_enemy = LAN
							alliance_with = ROOT
						}
					}
				}
			}
			else = {
				NOT = {
					any_country = {
						capital_scope = {
							province_id = 116
						}
						army_size = ROOT
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					OR = {
						alliance_with = LAN
						is_in_trade_league_with = LAN
						LAN = {
							has_opinion = {
								who = ROOT 
								value = 125
							}
						}
					}
				}
				add_country_modifier = {
					name = sie_rivalry_diminshed
					duration = 7300
				}
			}
			else = {
				add_country_modifier = {
					name = sie_archrival_florence
					duration = 18250
				}
			}
		}
	}

	sie_siena_papacy = {
		icon = reclaim_the_ducal_lands
		required_missions = {
			sie_old_rivalries
		}
		position = 3
		trigger = {
			if = {
				limit = { exists = PAP }
				OR = {
					alliance_with = PAP
					PAP = {
						has_opinion = {
							who = ROOT 
							value = 125
						}
					}
					is_rival = PAP
				}
			}
			else = {
				stability = 1
			}
		}
		effect = {
			if = {
				limit = {
					PAP = {
						has_opinion = {
							who = ROOT 
							value = 125
						}
					}
				}
				add_country_modifier = {
					name = sie_supporting_papacy
					duration = 7300
				}
			}
			else = {
				add_country_modifier = {
					name = sie_resisting_papal_influence
					duration = 7300
				}
			}
		}
	}

	sie_state_church = {
		icon = geneva_conventions
		required_missions = {
			sie_siena_papacy
		}
		position = 5
		trigger = {
			OR = {
				temple = 2
				cathedral = 2
			}
			if = {
				limit = { religion = catholic }
				num_of_cardinals = 1
				papal_influence = 50
			}
			else_if = {
				limit = { has_estate = estate_church }
				estate_loyalty = {
					estate = estate_church
					loyalty = 65
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = sie_influencing_church
				duration = 7300
			}
		}
	}

	sie_sienese_pope = {
		icon = defender_of_the_faith
		required_missions = { sie_state_church }
		position = 6
		trigger = {
			if = {
				limit = { religion = catholic }
				OR = {
					is_papal_controller = yes
					num_of_cardinals = 1
				}
			}
			else = {
				OR = {
					owns = 118
					NOT = {
						italy_region = {
							owned_by = PAP
						}
					}
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = sie_pope_siena
				duration = -1
			}
			if = {
				limit = { religion = catholic }
				country_event = {
					id = ME_Siena_Events.1
				}
				add_country_modifier = {
					name = sie_control_papacy
					duration = 7300
				}
			}
			else = {
				add_country_modifier = {
					name = sie_chair_st_peter
					duration = 7300
				}
			}
		}
	}
}
