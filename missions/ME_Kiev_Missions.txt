Kiev_missions_3 = {
	slot = 3
	generic = no
	ai = yes
	potential = {
		tag = KIE
		OR = {
			has_country_flag = use_missions_expanded_missions_instead_of_paradox
			NOT = { has_dlc = "Third Rome" }
		}
	}
	has_country_shield = yes

	Feed_Kiev = {
		icon = mission_dominate_home_trade_node
		required_missions = { }
		position = 1
		provinces_to_highlight = {
			province_id = 280
			OR = {
				NOT = { owned_by = ROOT }
				AND = {
					has_dlc = "Common Sense"
					NOT = { development = 16 }
				}
			}
		}
		trigger = {
			owns = 280
			if = {
				limit = {
					has_dlc = "Common Sense"
				}
				280 = {
					development = 16
				}
			}
			else = {
				stability = 2
				adm_power = 200
			}
		}
		effect = {
			if = {
				limit = {
					has_dlc = "Common Sense"
				}
				add_adm_power = 100
			}
			else = {
				add_adm_power = -200
				280 = {
					add_base_tax = 2
					add_base_production = 2
					add_base_manpower = 2
				}
			}
			add_legitimacy_equivalent = { value = 10 }
			add_country_modifier = {
				name = "Well_Fed"
				duration = 7300
			}
			add_estate_loyalty = {
				estate = estate_burghers
				loyalty = 10
			}
		}
	}
