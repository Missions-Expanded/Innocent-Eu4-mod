		provinces_to_highlight = {
			province_id = 78
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		position = 8
		
		trigger = {
			78 = {
				owned_by = ROOT
			}
			OR = {
				NOT = { num_of_electors = 7 }
				is_elector = yes
			}
			will_back_next_reform = yes
		}
		effect = {
			IF = {
				limit = { is_elector = yes }
				add_country_modifier = {
					name = "elector_claimant"
					duration = 3650
				}
			}
			ELSE = {
				elector = yes
			}
		}
	}
	
	frn_elector = {
		icon = mission_empire	
		required_missions = {
			frn_mainz
		}
		position = 9
		
		trigger = {
			is_elector = yes
			will_back_next_reform = yes

				has_opinion = {
					who = ROOT
					value = 100
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "frn_mended_relations"

			}
