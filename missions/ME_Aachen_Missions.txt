	aachen_grows = {
		icon = me_mission_aachen_city_view
		required_missions = { }
		position = 7
		trigger = {
			grown_by_development = 20
		}

			lower_rhineland_area = {
				limit = {
					NOT = { owned_by = ROOT }

			}
		}
	}

	aachen_lower_rheinland_ambitions_trier = {
		icon = me_mission_aachen_ambitions_trier
		required_missions = { aachen_grows }
		position = 8
		provinces_to_highlight = {
			province_id = 80
			NOT = { owned_by = ROOT }
		}
		trigger = {
			if = {
				limit = {
					80 = {
						NOT = { is_core = ROOT }
					}
				}
				adm_power = 200
			}
			80 = {
				owned_by = ROOT
			}
		}
		effect = {
			if = {
				limit = {
					80 = {
						NOT = { is_core = ROOT }
					}
				}
				add_adm_power = -200
				80 = {
					add_core = ROOT
				}
			}
			else = {
				add_prestige = 10
			}
		}
	}

	aachen_take_pfalz = {
