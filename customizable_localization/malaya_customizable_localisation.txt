defined_text = {
    name = GetGInvestmentStatus
	random = no

    text = {
        localisation_key = ME_yes
        trigger = {
			has_country_flag = MSA_invest_in_production
			check_variable = {
				which = MSA_foreign_investment_amount
				value = 20
			}
			OR = {
				NOT = { has_country_flag = MSA_investment_g_cooldown }
				had_country_flag = {
					flag = MSA_investment_g_cooldown
					days = 5475
				}
			}
			if = {
				limit = {
					is_expanded_mod_active = { mod = trade_goods }
				}
				any_owned_province = {
					OR = {
						trade_goods = spices
						trade_goods = cloves
						trade_goods = nutmeg
						trade_goods = vanilla
					}
					NOT = { has_province_modifier = MSA_enhanced_spice_production }
				}
			}
			else = {
				any_owned_province = {
					OR = {
						trade_goods = spices
						trade_goods = cloves
					}
					NOT = { has_province_modifier = MSA_enhanced_spice_production }
				}
			}
        }
    }
    text = {
        localisation_key = ME_no
        trigger = {
			always = yes
        }
    }
}
