estate_maya = {
	icon = 11

	color = { 79 162 43 }

	# If true, country will get estate
	trigger = {
		has_country_flag = estate_maya_enabled
	}

	# These scale with loyalty & power
	country_modifier_happy = {
		adm_tech_cost_modifier = -0.1
		core_creation = -0.15
		embracement_cost = -0.15
	}
	country_modifier_neutral = {
		core_creation = -0.1
		embracement_cost = -0.1
	}
	country_modifier_angry = {
		adm_tech_cost_modifier = 0.05
		core_creation = 0.1
		embracement_cost = 0.1
	}

	land_ownership_modifier = {
		global_institution_spread = 0.2
	}

	province_independence_weight = {
		factor = 1
	}

	# Influence modifiers
	base_influence = 20
	influence_modifier = {
		desc = EST_VAL_OVER_95_INFLUENCE
		trigger = {
			estate_influence = {
				estate = estate_maya
				influence = 95
			}
		}
		influence = 5
	}
	
	influence_modifier = {
		desc = EST_VAL_AHEAD_IN_ANY_TECH
		trigger = {
			all_country = {
				OR = {
					NOT = { adm_tech = ROOT }
					NOT = { dip_tech = ROOT }
					NOT = { mil_tech = ROOT }
				}
			}
			any_country = {
				OR = {
					adm_tech = ROOT
					dip_tech = ROOT
					mil_tech = ROOT
				}
			}
		}
		influence = 5
	}
	influence_modifier = {
		desc = EST_VAL_AHEAD_IN_ALL_TECH
		trigger = {
			all_country = {
				NOT = { adm_tech = ROOT }
				NOT = { dip_tech = ROOT }
				NOT = { mil_tech = ROOT }
			}
		}
		influence = 10
	}
	influence_modifier = {
		desc = EST_VAL_LOW_LEGITIMACY_EQUIVALENT
		trigger = {
			NOT = { legitimacy_equivalent = 50 }
		}
		influence = 10
	}

	# Loyalty modifiers
	loyalty_modifier = {
		desc = EST_VAL_MAYA_ESTATE_IN_10_MESOAMERICAN_RELIGION_PROVS
		trigger = {
			num_of_owned_provinces_with = {
				value = 10
				has_estate = estate_maya
				religion = mesoamerican_religion
			}
		}
		loyalty = 5
	}
	loyalty_modifier = {
		desc = EST_VAL_COMPETED_INNOVATIVENESS_IDEAS
		trigger = {
			full_idea_group = innovativeness_ideas
		}
		loyalty = 5
	}
	loyalty_modifier = {
		desc = EST_VAL_COMPETED_ADMINISTRATIVE_IDEAS
		trigger = {
			full_idea_group = administrative_ideas
		}
		loyalty = 5
	}
	loyalty_modifier = {
		desc = EST_VAL_OWN_INSTITUTION_ORIGIN
		trigger = {
			num_of_owned_and_controlled_institutions = 1
		}
		loyalty = 10
	}
	loyalty_modifier = {
		desc = EST_VAL_MAYA_ESTATE_HAS_HIGH_INFLUENCE
		trigger = {
			has_country_modifier = estate_maya_high_influence
		}
		loyalty = -99
	}

	influence_from_dev_modifier = 0.5

	privileges = {
		estate_maya_honor_kuhun

	}

	}
