me_hab_germanize = {
    icon = GFX_holy_order_jesuits
    trigger = {
        has_country_flag = me_hab_germanization_flag
		OR = {
			NOT = { has_country_flag = me_hab_germanized_province }
			had_country_flag = { me_hab_germanized_province days = 7300 }
		}
    }
	
    cost = 200
    cost_type = adm_power
	
    per_province_effect = {
		spawn_rebels = {
			type = nationalist_rebels
			size = 3
		}
        change_culture = ROOT
		owner = {
			add_absolutism = 1
			clr_country_flag = me_hab_germanized_province
			set_country_flag = me_hab_germanized_province
		}
    }
    per_province_abandon_effect = {
        add_base_manpower = -3
		add_base_production = -3
		add_base_tax = -3
    }
    modifier = {
        local_unrest = 2
    }
    ai_priority = {
        factor = 0 #Let us not ruin cultures now
    }
}
