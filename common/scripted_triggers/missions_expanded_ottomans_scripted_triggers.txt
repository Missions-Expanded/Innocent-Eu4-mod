me_check_slaves_variable = {
	if = {
		limit = {
			check_variable = {
				which = ME_TUR_slave_variable
				value = 0
			}
			NOT = {
				check_variable = {
					which = ME_TUR_slave_variable
					value = 25
				}
			}
		}
		any_owned_province = {
			trade_goods = slaves
			OR = {
				has_province_modifier = ME_TUR_slavery_1
				has_province_modifier = ME_TUR_slavery_2
				has_province_modifier = ME_TUR_slavery_3
			}
		}
	}
	else_if = {
		limit = {
			check_variable = {
				which = ME_TUR_slave_variable
				value = 25
				}
			NOT = {
				check_variable = {
					which = ME_TUR_slave_variable
					value = 50
				}
			}
		}
		any_owned_province = {
			trade_goods = slaves
			OR = {
				has_province_modifier = ME_TUR_slavery_0
				has_province_modifier = ME_TUR_slavery_2
				has_province_modifier = ME_TUR_slavery_3
			}
		}
	}
	else_if = {
		limit = {
			check_variable = {
				which = ME_TUR_slave_variable
				value = 50
				}
			NOT = {
				check_variable = {
					which = ME_TUR_slave_variable
					value = 75
				}
			}
		}
		any_owned_province = {
			trade_goods = slaves
			OR = {
				has_province_modifier = ME_TUR_slavery_0
				has_province_modifier = ME_TUR_slavery_1
				has_province_modifier = ME_TUR_slavery_3
			}
		}
	}
	else_if = {
		limit = {
			check_variable = {
				which = ME_TUR_slave_variable
				value = 75
				}
		}
		any_owned_province = {
			trade_goods = slaves
			OR = {
				has_province_modifier = ME_TUR_slavery_0
				has_province_modifier = ME_TUR_slavery_1
				has_province_modifier = ME_TUR_slavery_2
			}
		}
	}
}
