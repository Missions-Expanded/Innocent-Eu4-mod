				remove_province_modifier = can_quebecois_discontent
				add_province_modifier = {
					name = can_quebecois_content
					duration = -1
				}
			}
		}
	}

	option = {
		name = ME_Canada_Events.200.OPT1

		ai_chance = {
			factor = 100
		}
	}
