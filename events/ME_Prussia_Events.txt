country_event = {
	id = ME_Prussia_Events.1
	title = ME_Prussia_Events.1.t
	desc = ME_Prussia_Events.1.d
	picture = ELECTION_REPUBLICAN_eventPicture
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	
	option = {
		name = ME_Prussia_Events.1.a
		
	
		MAG = {
			country_event = {
				id = ME_Prussia_Events.2
			}
		}
		custom_tooltip = BRA_if_mag_accepts
		tooltip = {
			ROOT = {
				add_years_of_income = -2.0
			}
			MAG = {
				define_ruler = {
					name = "Albert"
					dynasty = "von Hohenzollern"
					dip = 2
					adm = 1
					age = 23
				}
				if = {
					limit = {
						is_expanded_mod_active   = { mod = governments }
					}
					add_government_reform = dynastic_theocracy_reform
				}
				custom_tooltip = BRA_recurring_hohenzollerns
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 5
				NOT = {
					num_of_loans = 1
				}
			}
			modifier = {
				factor = 0
				is_at_war = yes
			}
			modifier = {
				factor = 2
				DIP = 3
			}
			modifier = {
				factor = 3
				DIP = 4
			}
			modifier = {
				factor = 100
				BRA = {
					DIP = 6
				}
			}
		}
	}
	option = {
		name = ME_Prussia_Events.1.b
		ai_chance = {
			factor = 1
		}
		
		random_owned_province = {
			if = {
				limit = { has_building = temple }
				remove_building = temple
				add_building = cathedral
			}
			else_if = {
				limit = { has_building = cathedral }
				add_base_tax = 1
				add_base_production = 1
			}
			else = {
				add_building = temple
			}
		}
	}
}
