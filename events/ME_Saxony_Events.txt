country_event = {
	id = ME_Saxony_Events.3
	title = ME_Saxony_Events.EVTNAME3
	desc = ME_Saxony_Events.EVTDESC3
	picture = DIPLOMACY_eventPicture

	major = yes
	fire_only_once = yes

	trigger = {
		has_country_flag = saxony_claim_east_frankish_heritage_completed
		OR = {
			AND = {
				exists = FRX
				FRX = {
					is_subject = no
				}
			}
			AND = {
				exists = FRA
				FRA = {
					is_subject = no
					has_country_flag = restored_the_frankish_realm
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 60
	}

	option = {
		name = ME_Saxony_Events3.EVTOPT1

		ai_chance = {
			factor = 0.25

			modifier = {
				factor = 0.1
				OR = {
					AND = {
						exists = FRA
						FRA = {
							is_subject = no
							has_country_flag = restored_the_frankish_realm
						}
						OR = {
							alliance_with = FRA
							dynasty = FRA
							marriage_with = FRA
							historical_friend_with = FRA
						}
					}
					AND = {
						NOT = {
							exists = FRA
							FRA = {
								is_subject = no
								has_country_flag = restored_the_frankish_realm
							}
						}
						OR = {
							alliance_with = FRX
							dynasty = FRX
							marriage_with = FRX
							historical_friend_with = FRX
						}
					}
				}
			}
		}

		if = {
			limit = {
				calc_true_if = {
					all_country = {
						is_enemy = ROOT
					}
					amount = 3
				}
				any_rival_country = {
					NOT = {
						any_country = {
							NOT = { tag = PREV }
							is_enemy = ROOT
							NOT = { total_development = PREV }
						}
					}
				}
			}
			random_rival_country = {
				limit = {
					NOT = {
						any_country = {
							NOT = { tag = PREV }
							is_enemy = ROOT
							NOT = { total_development = PREV }
						}
					}
				}
				ROOT = {
					remove_rival = PREV
				}
			}
		}
		if = {
			limit = {
				exists = FRA
				FRA = {
					is_subject = no
					has_country_flag = restored_the_frankish_realm
				}
			}
			add_rival = FRA
			if = {
				limit = {
					OR = {
						historical_friend_with = FRA
						FRA = {
							historical_friend_with = ROOT
						}
					}
				}
				remove_historical_friend = FRA
				FRA = {
					remove_historical_friend = ROOT
				}
			}
		}
		else = {
			add_rival = FRX
			if = {
				limit = {
					OR = {
						historical_friend_with = FRX
						FRX = {
							historical_friend_with = ROOT
						}
					}
				}
				remove_historical_friend = FRX
				FRX = {
					remove_historical_friend = ROOT
				}
			}
		}
	}

	option = {
		name = ME_Saxony_Events3.EVTOPT2

		highlight = yes

		trigger = {
			if = {
				limit = {
					exists = FRA
					FRA = {
						is_subject = no
						has_country_flag = restored_the_frankish_realm
					}
				}
				OR = {
					alliance_with = FRA
					dynasty = FRA
					marriage_with = FRA
				}
			}
			else = {
				OR = {
					alliance_with = FRX
					dynasty = FRX
					marriage_with = FRX
				}
			}
		}

		ai_chance = { factor = 0.5 }

		if = {
			limit = {
				exists = FRA
				FRA = {
					is_subject = no
					has_country_flag = restored_the_frankish_realm
				}
			}
			add_casus_belli = {
				target = FRA
				type = cb_restore_personal_union
			}
		}
		else = {
			add_casus_belli = {
				target = FRX
				type = cb_restore_personal_union
			}
		}
	}

	option = {
		name = ME_Saxony_Events3.EVTOPT3

		ai_chance = { factor = 0.25 }

		add_prestige = -10
	}
}
