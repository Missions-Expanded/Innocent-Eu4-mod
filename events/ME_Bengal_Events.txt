country_event = { #Standard Currency of Trade
	id = ME_Bengal_Events.2
	title = ME_Bengal_Events.2.t
	desc = ME_Bengal_Events.2.d
	picture = TRADEGOODS_eventPicture
	is_triggered_only = yes

	option = {
		name = ME_Bengal_Events.2.opt1
		1248 = {
			add_permanent_province_modifier = {
				name = bng_cowry_shell_exports_modifier
				duration = -1
			}
			if = {
				limit = { 
					is_expanded_mod_active = { mod = "trade_goods" } 
				}
				change_trade_goods = pearls
			}
			else = {
				change_trade_goods = gems
			}
		}
		if = {
			limit = {
				AND = {
					1248 = { owned_by = ROOT }
					DGL = {
						NOT = { exists = yes }
					}
				}
			}
			1248 = { create_vassal = DGL }
		}
		custom_tooltip = BNG_COWRY_TRADE_TT
		set_country_flag = bng_shell_currency
	}
	option = {
		name = ME_Bengal_Events.2.opt2
		1248 = {
			add_permanent_province_modifier = {
				name = bng_cowry_shell_exports_modifier
				duration = -1
			}
		}
		if = {
			limit = {
				NOT = {
					1248 = { owned_by = ROOT }
				}
			}
			add_years_of_income = 0.5
			add_dip_power = 50
		}
	}
	option = {
		name = ME_Bengal_Events.2.opt3
		add_mercantilism = 5
		1248 = {
			add_permanent_province_modifier = {
				name = bng_cowry_shell_exports_modifier
				duration = -1
			}
		}
		if = {
			limit = {
				1248 = { owned_by = ROOT }
			}
			1248 = {
				add_base_production = 1
				add_base_tax = 1
				set_local_autonomy = 10
			}
		}
		else = {
			1248 = { add_permanent_claim = ROOT }
		}
	}
}
