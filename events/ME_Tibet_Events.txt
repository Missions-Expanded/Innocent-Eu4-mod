namespace = ME_Tibet_Events

# The Dalai Lama takes control
country_event = {
	id = ME_Tibet_Events.1
	title = "flavor_tib.EVTNAME1"
	desc = "flavor_tib.EVTDESC1"
	picture = RELIGION_eventPicture

	is_triggered_only = yes
	
	option = {
		trigger = {
			NOT = { has_reform = celestial_empire }
		}				
		name = "flavor_tib.EVTOPTA1"
		ai_chance = { factor = 100 }
		if = {
			limit = {
				NOT = {
					religion = vajrayana
				}
			}
			change_religion = vajrayana
		}
		if = {
			limit = {
				NOT = {
					government = theocracy
				}
			}
			change_government_to_theocracy = yes
			add_government_reform = leading_clergy_reform
			#set_legacy_government = theocratic_government
		}
		else = {
			add_devotion = 20
		}
	}
	option = {
		name = "flavor_tib.EVTOPTB1"
		ai_chance = { factor = 0 }
		add_stability_or_adm_power = yes
	}
}

# Tibetan relations with the Hordes by Daniel Tolman
country_event = {
	id = ME_Tibet_Events.2
	title = "flavor_tib.EVTNAME2"
	desc = "flavor_tib.EVTDESC2"
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			set_country_flag = tibetan_government_flag # trigger for tibet missions if tag is Khoshuud, Dzungar, or Khalsa
		}
	}
	
	option = { # Improve relations with the hordes
		name = "flavor_tib.EVTOPTA2"
		ai_chance = { factor = 100 }
		every_country = {
			limit = {
				OR = {
					primary_culture = altaic_new 
					culture_group = altaic
				}
				has_reform = steppe_horde
			}
			add_opinion = {
				who = ROOT
				modifier = tibetan_steppe_relations
			}
		}
	}
	option = { # Khalkha Mongols
		name = "flavor_tib.EVTOPTB2"
		ai_chance = { factor = 0 }
		trigger = {
			NOT = {
				exists = KLK
			}
			NOT = { has_reform = celestial_empire }
			was_never_end_game_tag_trigger = yes
		}
		add_stability = -2
		change_tag = KLK
		change_government = tribal
		add_government_reform = steppe_horde
		change_primary_culture = khalkha
		if = {
			limit = {
				has_global_flag = sinicized_tibetan_flag
			}
			add_accepted_culture_or_dip_power = { culture = tibetan_new }
		}
		else = {
			add_accepted_culture_or_dip_power = { culture = tibetan }
		}
		if = {
			limit = {
				is_year = 1527
				NOT = { is_year = 1582 }
			}
			define_ruler = {
				name = "Altan"
				dynasty = "Borjigin"
				culture = khalkha
				adm = 4
				dip = 4
				mil = 5
			}
		}
		else = {
			define_ruler = {
				dynasty = "Borjigin"
				culture = khalkha
			}
		}
	}
	option = { # Khoshuud Oirats
		name = "flavor_tib.EVTOPTC2"
		ai_chance = { factor = 0 }
		trigger = {
			NOT = {
				exists = KSD
			}
			NOT = { has_reform = celestial_empire }
			was_never_end_game_tag_trigger = yes
		}
		add_stability = -2
		change_tag = KSD
		change_primary_culture = oirats
		if = {
			limit = {
				has_global_flag = sinicized_tibetan_flag
			}
			add_accepted_culture_or_dip_power = { culture = tibetan_new }
		}
		else = {
			add_accepted_culture_or_dip_power = { culture = tibetan }
		}
		change_government = tribal
		add_government_reform = steppe_horde
		if = {
			limit = {
				is_year = 1600
				NOT = { is_year = 1655 }
			}
			define_ruler = {
				name = "Gushri"
				dynasty = "Khoshut"
				culture = oirats
				adm = 2
				dip = 4
				mil = 5
			}
		}
		else = {
			define_ruler = {
				dynasty = "Khoshut"
				culture = oirats
			}
		}
	}
	option = { # Dzungar Oirats
		name = "flavor_tib.EVTOPTD2"
		ai_chance = { factor = 0 }
		trigger = {
			is_year = 1634
			NOT = {
				exists = ZUN
			}
			NOT = { has_reform = celestial_empire }
			was_never_end_game_tag_trigger = yes
		}
		add_stability = -2
		change_tag = ZUN
		change_primary_culture = oirats
		if = {
			limit = {
				has_global_flag = sinicized_tibetan_flag
			}
			add_accepted_culture_or_dip_power = { culture = tibetan_new }
		}
		else = {
			add_accepted_culture_or_dip_power = { culture = tibetan }
		}
		change_government = tribal
		add_government_reform = steppe_horde
		define_ruler = {
			dynasty = "Choros"
			culture = oirats
		}
	}
}
