country_event = {  # The Finnish Union
	id = ME_Finland_Events.4
	title = ME_Finland_Events.4.title
	desc = ME_Finland_Events.4.desc
	# picture =

	fire_only_once = yes
	mean_time_to_happen = { months = 120 }

	trigger = {
		tag = FIN
		owns = 12
		owns = 17
		owns = 1
		owns = 28
		NOT = {
			calc_true_if = {
				amount = 5
				all_province = {
					region = scandinavia_region
					NOT = { owned_by = FIN }
				}
			}
		}
		primary_culture = finnish
	}

	option = {
		name = ME_Finland_Events.4.a
		ai_chance = { factor = 50 }
		override_country_name = "Finnish Union"
		set_government_rank = 3
		add_prestige = 15
		add_country_modifier = {
			name = "centralization_modifier"
			duration = 5475
		}
		scandinavia_region = {
			limit = {
				NOT = { owned_by = ROOT }
				NOT = { is_core = ROOT }
			}
			add_permanent_claim = ROOT
		}
		set_country_flag = formed_scandinavia_flag
	}

	option = {
		name = ME_Finland_Events.4.b
		ai_chance = { factor = 50 }
		add_stability = 2
	}
}
