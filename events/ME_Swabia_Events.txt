country_event = {
	id = ME_Swabia_Events.1 #Hohenstaufen
	title = ME_Swabia_Events.1.t
	desc = ME_Swabia_Events.1.d
	picture = HRE_eventPicture

	fire_only_once = yes

	trigger = {
		tag = SWA
		government = monarchy
		religion = catholic
	}

	mean_time_to_happen = {
		months = 60
	}

	option = {
		name = ME_Swabia_Events.1.a

		ai_chance = { factor = 0 }

		set_country_flag = SWA_Barbarossa
		custom_tooltip = SWA_Barbarossa_tooltip
		hidden_effect = {
			swap_non_generic_missions = yes
		}
		add_country_modifier = {
			name = SWA_Barbarossa_legacy
			duration = 36500
		}
		add_legitimacy_equivalent = { amount = -5 republican_tradition = -3 }
		every_owned_province = {
			add_unrest = 10
		}
		set_dynasty = "von Hohenstaufen"
	}

	option = {
		name = ME_Swabia_Events.1.b

		ai_chance = { factor = 100 }

		add_prestige = 5
	}
}

country_event = {
	id = ME_Swabia_Events.2 #reform center
	title = ME_Swabia_Events.2.t
	desc = ME_Swabia_Events.2.d
	picture = HRE_eventPicture

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = ME_Swabia_Events.2.a #in capital

		trigger = {
			capital_scope = {
				religion = catholic
			}
		}

		ai_chance = { factor = 20 }

		capital_scope = {
			add_reform_center = catholic
		}
	}

	option = {
		name = ME_Swabia_Events.2.b

		trigger = {
			any_province = {
				is_part_of_hre = yes
				religion = catholic
				any_neighbor_province = {
					OR = {
						religion = protestant
						religion = reformed
						religion = anglican
					}
				}
			}
		}

		ai_chance = { factor = 20 }

		hidden_effect = {
			random_province = {
				limit = {
					is_part_of_hre = yes
					religion = catholic
					any_neighbor_province = {
						OR = {
							religion = protestant
							religion = reformed
							religion = anglican
						}
					}
				}
				add_reform_center = catholic
			}
		}
		custom_tooltip = SWA_tooltip_for_reform_center
	}

	option = {
		name = ME_Swabia_Events.2.c #Primas Germaniae

		trigger = {
			any_country = {
				has_country_modifier = primas_germaniae
			}
		}

		ai_chance = { factor = 20 }

		random_country = {
			limit = {
				has_country_modifier = primas_germaniae
			}
			capital_scope = {
				add_reform_center = catholic
			}
		}
	}

	option = {
		name = ME_Swabia_Events.2.e #Legatus Natus

		trigger = {
			any_country = {
				has_country_modifier = legatus_natus
			}
		}

		ai_chance = { factor = 20 }

		random_country = {
			limit = {
				has_country_modifier = legatus_natus
			}
			capital_scope = {
				add_reform_center = catholic
			}
		}
	}

	option = {
		name = ME_Swabia_Events.2.f #Pope

		trigger = {
			exists = PAP
			PAP = {
				religion = catholic
				capital_scope = {
					religion = catholic
				}
			}
		}

		ai_chance = { factor = 20 }

		PAP = {
			capital_scope = {
				add_reform_center = catholic
			}
		}
	}
}

country_event = {
	id = ME_Swabia_Events.3 #Invitation
	title = ME_Swabia_Events.3.t
	desc = ME_Swabia_Events.3.d
	picture = HRE_eventPicture

	is_triggered_only = yes

	option = {
		name = ME_Swabia_Events.3.a #accept

		ai_chance = {
			factor = 20
			modifier = {
				factor = 5
				variable_arithmetic_trigger = {
					export_to_variable = {
						which = our_cities
						value = trigger_value:num_of_cities
					}
					export_to_variable = {
						which = their_cities
						value = trigger_value:num_of_cities
						who = FROM
					}
					multiply_variable = { our_cities = 3 }
					check_variable = { which = their_cities which = our_cities }
				}
				variable_arithmetic_trigger = {
					export_to_variable = {
						which = our_army
						value = trigger_value:army_size
					}
					export_to_variable = {
						which = their_army
						value = trigger_value:army_size
						who = FROM
					}
					multiply_variable = { our_army = 3 }
					check_variable = { which = their_army which = our_army }
				}
			}
			modifier = {
				factor = 0.1
				variable_arithmetic_trigger = {
					export_to_variable = {
						which = our_cities
						value = trigger_value:num_of_cities
					}
					export_to_variable = {
						which = their_cities
						value = trigger_value:num_of_cities
						who = FROM
					}
					multiply_variable = { their_cities = 3 }
					check_variable = { which = their_cities which = our_cities }
				}
				variable_arithmetic_trigger = {
					export_to_variable = {
						which = our_army
						value = trigger_value:army_size
					}
					export_to_variable = {
						which = their_army
						value = trigger_value:army_size
						who = FROM
					}
					multiply_variable = { their_army = 3 }
					check_variable = { which = their_army which = our_army }
				}
			}
		}

		every_owned_province = {
			limit = {
				continent = europe
				is_part_of_hre = no
			}
			set_in_empire = yes
		}
	}

	option = {
		name = ME_Swabia_Events.3.b #refuse

		ai_chance = {
			factor = 100
			modifier = {
				factor = 0.2
				NOT = { num_of_cities = 2 }
			}
			modifier = {
				factor = 0.5
				NOT = { num_of_cities = 5 }
			}
		}

		add_prestige = 5
	}
}
