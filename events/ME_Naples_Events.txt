			owned_by = ROOT

	}

		hidden_effect = {

		}

	}

		name = ME_Naples_Events.2.OPT2

		ai_chance = {
			factor = 25
		}
		trigger = {
			is_expanded_mod_active = { mod = trade_goods }
		}

			change_trade_goods = lumber

				name = mr_piles_supplies

				33 = { add_base_tax = 2 }

			}

	}

	option = {
		name = ME_Naples_Events.2.OPT3

		ai_chance = {
			factor = 25
		}

		2983 = {
			change_trade_goods = paper

				name = nap_charta_bambagina
				duration = -1
			}

	}

	option = {
		name = ME_Naples_Events.2.OPT4

		ai_chance = {
			factor = 25
		}
		trigger = {
			is_expanded_mod_active = { mod = trade_goods }

#The Throne of Provence for Provence
country_event = {
	id = ME_Naples_Events.3
	title = ME_Naples_Events.3.T
	desc = ME_Naples_Events.3.D
	picture = COURT_eventPicture

	is_triggered_only = yes
	major = yes
	major_trigger = {
		tag = event_target:major_notification_receiver
	}

	immediate = {
		hidden_effect = {
			FROM = {
				save_event_target_as = major_notification_receiver
			}
		}
	}

	option = {
		name = ME_Naples_Events.3.OPT1
		ai_chance = {
			factor = 20
			modifier = {
				factor = 3
				army_strength = {
					who = FROM
					value = 0.25
				}
			}
			modifier = {
				factor = 2
				army_strength = {
					who = FROM
					value = 0.50
				}
			}
			modifier = {
				factor = 1.5
				FROM = {
					diplomatic_reputation = 3
				}
			}
			modifier = {
				factor = 2
				NOT = { legitimacy = 50 }
			}
			modifier = {
				factor = 2
				NOT = { stability = 1 }
			}
		}

		trigger = {
			hidden_trigger = {
				is_subject = no
				OR = {
					AND = {
						valid_for_personal_unions_trigger = yes
						FROM = {
							valid_for_personal_unions_trigger = yes
						}
					}
					AND = {
						is_expanded_mod_active = { mod = subjects }
						FROM = {
							can_create_subject_type = {
								subject = ROOT
								subject_type = brother_kingdom
							}
						}
					}
				}
			}
		}

		set_dynasty = FROM
		FROM = {
			if = {
				limit = {
					is_expanded_mod_active = { mod = subjects }
					FROM = {
						can_create_subject_type = {
							subject = ROOT
							subject_type = brother_kingdom
						}
					}
				}
				create_subject = {
					subject_type = brother_kingdom
					subject = ROOT
				}
			}
			else = {
				create_subject = {
					subject_type = personal_union
					subject = ROOT
				}
			}
		}
	}

	option = {
		name = ME_Naples_Events.3.OPT2
		ai_chance = {
			factor = 80
		}

		add_casus_belli = {
			type = cb_insult
			target = FROM
		}
		every_owned_province = {
			limit = {
				NOT = { is_core = FROM }
			}
			add_permanent_claim = FROM
		}
		FROM = {
			add_casus_belli = {
				type = cb_restore_personal_union
				months = 300
				target = ROOT
			}
		}
	}
}
