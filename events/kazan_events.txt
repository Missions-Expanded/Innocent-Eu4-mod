country_event = { #question of perm
	id = KAZ_mission_events.4
	title = KAZ_mission_events.4.t
	picture = BORDER_TENSION_eventPicture
	desc = KAZ_mission_events.4.d

	fire_only_once = yes
	
	is_triggered_only = yes
	
	option = {
		name = KAZ_mission_events.4.a
		trigger = {
			exists = PRM
		}
		PRM = {
			every_owned_province = {
				cede_province = KAZ
				add_province_modifier = {
					name = kaz_perm
					duration = 9125
				}
			}
		}
	}
	option = {
		name = KAZ_mission_events.4.b
		trigger = {
			exists = PRM
		}
		PRM = {
			grant_independence = yes
		}
		create_alliance = PRM
	}
	option = {
		name = KAZ_mission_events.4.c
		trigger = {
			exists = PRM
		}
		create_march = PRM
	}
	option = {
		name = KAZ_mission_events.4.d
		trigger = {
			exists = PRM
		}
		create_subject = {
			subject_type = tributary_state
			subject = PRM
		}
	}
	
	#in case perm doesnt exist
	option = {
		name = KAZ_mission_events.4.d
		trigger = {
			NOT = { exists = PRM }
		}
		perm_area = {
			add_devastation = 10
		}
		add_years_of_income = 0.5
	}
}
