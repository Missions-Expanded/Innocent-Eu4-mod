country_event = {
	id = ME_Malacca_Events.5 #Fall of Malacca
	title = ME_Malacca_Events.5.t
	desc = ME_Malacca_Events.5.d
	picture = CITY_DEVELOPMENT_eventPicture

	fire_only_once = yes

	trigger = {
		tag = MLC
		596 = {
			NOT = { owned_by = ROOT }
			owner = {
				capital_scope = {
					continent = europe
				}
			}
		}
		owns = 597
		owns = 595
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = ME_Malacca_Events.5.a

		ai_chance = {
			factor = 1
		}

		595 = {
			cede_province = PRK
		}
		PRK = {	set_capital = 595 }
		597 = {
			cede_province = JOH
		}
		JOH = {	set_capital = 597 }
		every_owned_province = {
			ROOT = { set_capital = PREV }
			if = {
				limit = {
					ROOT = {
						variable_arithmetic_trigger = {
							export_to_variable = {
								variable_name = JOH_Distance
								value = capital_distance
								who = JOH
							}
							export_to_variable = {
								variable_name = PRK_Distance
								value = capital_distance
								who = PRK
							}
							check_variable = {
								which = PRK_Distance
								which = JOH_Distance
							}
						}
					}
				}
				cede_province = JOH
				if = {
					limit = {
						is_core = ROOT
					}
					add_core = JOH
				}
			}
			else = {
				cede_province = PRK
				if = {
					limit = {
						is_core = ROOT
					}
					add_core = PRK
				}
			}
		}
	}
}
