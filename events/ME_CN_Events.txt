		}

					always = yes

			}
		}

province_event = {  #Growth of city
	id = ME_CN_Events.10
	title = ME_CN_Events.10.t
	desc = ME_CN_Events.10.d
	picture = COLONIZATION_eventPicture

	trigger = {
		owner = {
			has_country_flag = CN_cool_trade_goods
		}
		if = {
					limit = {
						is_expanded_mod_active = { mod = development }
		   		}
				development = 36
			}
			else = {
				development = 18
			}
		NOT = { has_province_flag = CN_trade_good_event_happened }
		NOT = {
			OR = {
				trade_goods = gold
				trade_goods = glass
				trade_goods = paper
				trade_goods = silk
				trade_goods = cloth
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	immediate = {
		hidden_effect = {
			set_province_flag = CN_trade_good_event_happened
		}
	}

	option = { #Paper
		name = ME_CN_Events.10.a

		ai_chance = { factor = 40 }

		change_trade_goods = paper
	}

	option = { #Glass
		name = ME_CN_Events.10.b

		ai_chance = { factor = 40 }

		change_trade_goods = glass
	}

	option = { #cloth
		name = ME_CN_Events.10.c
		ai_chance = { factor = 20 }

		change_trade_goods = cloth
	}

	option = { #silk
		name = ME_CN_Events.10.e

		trigger = {
			culture_group = east_asian
		}

		ai_chance = { factor = 60 }

		change_trade_goods = silk
	}

	option = { #chinaware
		name = ME_CN_Events.10.f

		trigger = {
			culture_group = east_asian
		}

		ai_chance = { factor = 60 }

		change_trade_goods = chinaware
	}
	option = { #Nothing
		name = ME_CN_Events.10.g

		ai_chance = { factor = 60 }

		add_province_modifier = {
			name = CN_growth
			duration = -1
		}
	}
}
