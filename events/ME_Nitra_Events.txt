#Panslavism 1
province_event = {
	id = me_nitra.7
	title = me_nitra.7.t
	desc = me_nitra.7.d
	picture = CITY_VIEW_eventPicture
	
	hidden = yes
	
	trigger = {
		owner = {
			has_country_flag = slo_panslavism_flag
		}
		OR = {
			culture_group = west_slavic
			culture_group = east_slavic
			culture_group = south_slavic
			culture = slovak
		}
		NOT = {
			has_owner_accepted_culture = yes
		}
		NOT = {
			has_province_modifier = slo_panslavism_modifier
		}
	}
	
	option = {
		name = me_nitra.7.a
		add_province_modifier = {
			name = "slo_panslavism_modifier"
			duration = -1
		}
	}
	
}

#Panslavism Birth
country_event = {
	id = me_nitra.9
	title = me_nitra.9.t
	desc = me_nitra.9.d
	picture = CONQUEST_eventPicture
	
	is_triggered_only = yes
	
	major = yes
	
	option = {
		name = me_nitra.9.a
		custom_tooltip = SLO_PANSLAVISM
		set_country_flag = slo_panslavism_flag
		add_country_modifier = {
			name = "slo_blood_faith"
			duration = -1
		}
		custom_tooltip = SLO_SLAVIC_PERMA_CLAIM
		hidden_effect = {
			every_province = {
				limit = {
					OR = {
						culture_group = east_slavic
						culture_group = south_slavic
						culture_group = west_slavic
						culture = slovak
					}
					NOT = {
						OR = {
							owned_by = ROOT
							is_core = ROOT
							is_permanent_claim = ROOT
						}
					}
					
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
}
