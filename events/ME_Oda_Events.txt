country_event = { # The Rise of Oda Nobunaga
	id = ME_Oda_Events.1
	title = ME_Oda_Events.EVTNAME1
	desc = ME_Oda_Events.EVTDESC1
	picture = COURT_eventPicture

	fire_only_once = yes

	trigger = {
		tag = ODA
		mission_completed = ODA_western_chubu
		NOT = { started_in = 1534.6.23 }
	}

	mean_time_to_happen = { months = 12 }

	option = {
		name = ME_Oda_Events1.EVTOPT1

		ai_chance = { factor = 50 }

		tooltip = {
			kill_ruler = yes
			if = {
				limit = {
					has_heir = yes
				}
				kill_heir = {}
			}
		}
		hidden_effect = { remove_heir = {} }
		define_ruler = {
			name = "Nobunaga"
			dynasty = "Oda"
			adm = 5
			dip = 5
			mil = 6
			age = 20
			claim = 100
		}
		set_country_flag = ODA_Nobunaga
		set_ruler_flag = ODA_Nobunaga
		if = {
			limit = {
				has_dlc = "Rights of Man"
			}
			add_ruler_personality = just_personality
			add_ruler_personality = well_connected_personality
			add_ruler_personality = conqueror_personality
		}
	}

	option = {
		name = ME_Oda_Events1.EVTOPT2

		ai_chance = {
			factor = 50
		}

		add_prestige = 10
		add_stability_or_adm_power = yes
	}
}

	option = {
		name = ME_Oda_Events4.EVTOPT1

		ai_chance = {

				factor = 2

					}
				}
			}
		}

		add_country_modifier = {

				name = ODA_appeased_shogun
				duration = 7300
			}
		}
	}

	option = {
		name = ME_Oda_Events4.EVTOPT2

		ai_chance = {
			factor = 50

			modifier = {
				factor = 2
				NOT = {
					has_opinion = {
						who = FROM
						value = -100
					}
				}
				NOT = {
					FROM = {
						has_opinion = {
							who = ROOT
							value = -100
						}
					}
				}
			}
		}

		add_country_modifier = {
			name = ODA_quelling_daimyo

		}

				duration = 7300
			}
		}
	}
