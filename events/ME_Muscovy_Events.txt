#Shemyaka's Rebellion
country_event = {
	id = ME_Muscovy_Events.70
	title = ME_Muscovy_Events.70.t
	desc = ME_Muscovy_Events.70.d
	picture = CIVIL_WAR_eventPicture

	major = yes
	major_trigger = {
		any_known_country = { tag = FROM }
		any_known_country = { tag = GOL }
	}

	fire_only_once = yes

	mean_time_to_happen = { months = 6 }

	trigger = {
		tag = MOS
		exists = GOL
		owns = 1778
		is_at_war = no
	}

	after = { set_country_flag = shemyaka_rebellion }

	option = {
		name = ME_Muscovy_Events.70.a

		ai_chance = { factor = 19 }

		add_stability = -1

		add_opinion = {
			who = GOL
			modifier = muscovy_supported_pretender
		}
		reverse_add_opinion = {
			who = GOL
			modifier = muscovy_supported_pretender
		}

		kill_leader = { type = "Dmitry Shemiaka" }

		1778 = {
			if = {
				limit = {
					owner = { ai = yes }
				}
				spawn_rebels = {
					type = pretender_rebels
					size = 1
					win = yes
					friend = GOL
					leader = "Dmitry"
					leader_dynasty = "Shemyaka"
				}
			}
			else = {
				spawn_rebels = {
					type = pretender_rebels
					size = 2.5
					win = yes
					friend = GOL
					leader = "Dmitry"
					leader_dynasty = "Shemyaka"
				}
			}
		}
	}

	option = {
		name = ME_Muscovy_Events.70.b

		ai_chance = { factor = 1 }

		add_stability = -1

		hidden_effect = {
			kill_leader = { type = "Dmitry Shemiaka" }
		}
		define_ruler = {
			name = "Dmitry"
			dynasty = "Shemyaka"
			claim = 20
			adm = 1
			max_random_adm = 1
			dip = 5
			max_random_dip = 5
			mil = 3
			max_random_mil = 3
			male = yes
		}
		hidden_effect = {
			define_ruler_to_general = {
				fire = 3
				shock = 4
				manuever = 3
				siege = 0
			}
		}

		GOL = {
			if = {
				limit = { has_dlc = "Mandate of Heaven" }
				create_subject = {
					subject_type = tributary_state
					subject = ROOT
				}
			}
			else = { vassalize = ROOT }
		}

		add_liberty_desire = -100

		if = {
			limit = {
				any_subject_country = { dynasty = "Rurikovich" }
			}
			custom_tooltip = ME_Muscovy_Events.70.tta
		}
		hidden_effect = {
			every_subject_country = {
				limit = { dynasty = "Rurikovich" }
				if = {
					limit = { has_saved_event_target = war_leader }
					create_alliance = event_target:war_leader
				}
				else = { save_event_target_as = war_leader }
			}
		}
		if = {
			limit = { has_saved_event_target = war_leader }
			event_target:war_leader = {
				declare_war_with_cb = {
					who = ROOT
					casus_belli = cb_independence_war
				}
			}
		}

		add_opinion = {
			who = GOL
			modifier = muscovy_approves_prince
		}
		reverse_add_opinion = {
			who = GOL
			modifier = muscovy_approves_prince
		}

		GOL = {
			country_event = {
				id = ME_Muscovy_Events.71 #Muscovite Civil War
				days = 0
			}
		}

		capital_scope = {
			if = {
				limit = { has_saved_event_target = war_leader }
				spawn_rebels = {
					type = pretender_rebels
					size = 3
					win = yes
					friend = event_target:war_leader
					leader = "Vasiliy II Temny"
					leader_dynasty = "Rurikovich"
				}
			}
			else = {
				spawn_rebels = {
					type = pretender_rebels
					size = 3
					win = yes
					leader = "Vasiliy II Temny"
					leader_dynasty = "Rurikovich"
				}
			}
		}
	}
}
