#Sack of Spain
country_event = {
	id = ME_The_Two_Sicilies_Events.1

	mean_time_to_happen = {
		months = 1
	}

	trigger = {
		NOT = { has_country_flag = tts_sack_spain }
		OR = {
			tag = NAP
			tag = SIC
			tag = TTS
		}
		OR = {
			AND = {
				war_score_against = {
					who = SPA
					value = 75
				}
				SPA = {
					capital_scope = {
						controlled_by = ROOT
					}
				}
			}
			AND = {
				war_score_against = {
					who = CAS
					value = 75
				}
				CAS = {
					capital_scope = {
						controlled_by = ROOT
					}
				}
			}
			AND = {
				war_score_against = {
					who = ARA
					value = 75
				}
				ARA = {
					capital_scope = {
						controlled_by = ROOT
					}
				}
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_country_flag = tts_sack_spain
		}
	}

	option = {
		name = ME_Naples_Events.16.OPT1
		ai_chance = {
			factor = 100
		}

		add_prestige = 10
		add_years_of_income = 3
		if = {
			limit = { war_with = CAS }
			CAS = {
				add_years_of_income = -3
				capital_scope = {
					add_devastation = 20
					add_base_tax = -3
					add_base_production = -1
					add_base_manpower = -2
				}
			}
		}
		else_if = {
			limit = { war_with = ARA }
			ARA = {
				add_years_of_income = -3
				capital_scope = {
					add_devastation = 20
					add_base_tax = -3
					add_base_production = -1
					add_base_manpower = -2
				}
			}
		}
		else_if = {
			limit = { war_with = SPA }
			SPA = {
				add_years_of_income = -3
				capital_scope = {
					add_devastation = 20
					add_base_tax = -3
					add_base_production = -1
					add_base_manpower = -2
				}
			}
		}
		capital_scope = {
			add_base_tax = 3
			add_base_production = 1
			add_base_manpower = 2
		}
	}
}

#The Siblings
country_event = {
	id = ME_The_Two_Sicilies_Events.3
	title = ME_The_Two_Sicilies_Events.3.T
	desc = ME_The_Two_Sicilies_Events.3.D
	picture = ANGRY_MOB_eventPicture

	is_triggered_only = yes

	option = {
		name = ME_The_Two_Sicilies_Events.3.OPT1

		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				ai = yes
			}
		}

		hidden_effect = {
			catalonia_area = {
				limit = {
					NOT = { is_core = CAT }
				}
				add_core = CAT
			}
		}
		release = CAT
		CAT = {
			change_government = monarchy
			hidden_effect = {
				set_dynasty = FROM	
				change_religion = FROM
				add_treasury = 20
				add_yearly_manpower = 10
				add_yearly_sailors = 10
			}
		}
		if = {
			limit = {
				is_expanded_mod_active = { mod = subjects }
				can_create_subject_type = {
					subject = ROOT
					subject_type = brother_kingdom
				}
			}
			create_subject = {
				subject_type = brother_kingdom
				subject = CAT
			}
		}
		else = {
			create_subject = {
				subject_type = personal_union
				subject = CAT
			}
		}
	}

	option = {
		name = ME_The_Two_Sicilies_Events.3.OPT2

		ai_chance = {
			factor = 50
		}

		catalonia_area = {
			limit = { owned_by = ROOT }
			add_unrest = 5
			add_province_modifier = {
				name = nap_forced_subdjucation
				duration = 5475
			}
		}
	}
}

country_event = { #Restoring the Glory of Greece
	id = ME_The_Two_Sicilies_Events.4
	title = ME_The_Two_Sicilies_Events.4.T
	desc = ME_The_Two_Sicilies_Events.4.D
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = ME_The_Two_Sicilies_Events.4.OPT1

		trigger = {
			NOT = { exists = GRE }
			num_of_owned_provinces_with = {
				value = 3
				OR = {
					area = northern_greece_area
					area = macedonia_area
					area = morea_area
					area = thrace_area
				}
			}
		}
		
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				NOT = { num_of_free_diplomatic_relations = 1 }
			}
		}

		hidden_effect = {
			every_owned_province = {
				limit = {
					NOT = { is_core = GRE }
					culture_group = byzantine
				}
				add_core = GRE
			}
		}
		release = GRE
		GRE = {
			change_government = monarchy
			hidden_effect = {
				set_dynasty = ROOT
				add_treasury = 20
				add_yearly_manpower = 10
				add_yearly_sailors = 10
			}
		}
		if = {

					subject = ROOT

				}
			}
			create_subject = {
				subject_type = brother_kingdom
				subject = GRE
			}
		}
		else = { create_march = GRE }
		northern_greece_area = {
			limit = {
				NOT = { is_core = GRE }
			}
			add_core = GRE
			add_permanent_claim = GRE
		}
		macedonia_area = {
			limit = {
				NOT = { is_core = GRE }
			}
			add_core = GRE
			add_permanent_claim = GRE
		}

				NOT = { is_core = GRE }
			}
			add_core = GRE
			add_permanent_claim = GRE
		}		
		aegean_archipelago_area = {
			limit = {
				NOT = { is_core = GRE }
			}
			add_core = GRE
			add_permanent_claim = GRE
		}

				limit = { region = italy_region }

					name = nap_subject_slaves
					duration = 9125
				}
			}
		}
	}
