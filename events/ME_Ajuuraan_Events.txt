					muslim_idea_groups_expanded = 2

					christian_idea_groups_expanded = 2
				}
			}
			remove_idea_group = christian_idea_groups_expanded
			remove_idea_group = religious_ideas 
			remove_idea_group = muslim_idea_groups_expanded
