#Jews of Sicily
country_event = {
	id = Melva_Jew_Italy_Events.1
	title = Melva_Jew_Italy_Events.1.T
	desc = Melva_Jew_Italy_Events.1.D
	picture = RELIGIOUS_TURMOIL_eventPicture
	
	trigger = {
		NOT = { is_year = 1470 }
		OR = {
			western_sicily_area = {
				owned_by = ROOT
			}
			sicily_area = {
				owned_by = ROOT
			}
		}
		NOT = { has_country_flag = melva_jew_initial_event }
	}

	immediate = {
		hidden_effect = {
			set_country_flag = melva_jew_initial_event
			set_variable = {
				which = Melva_JEW_OPPOSE
				value = 0
			}
			set_variable = {
				which = Melva_JEW_APPROVE
				value = 0
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = Melva_Jew_Italy_Events.1.OPT1
		ai_chance = {
			factor = 50
		}

		custom_tooltip = Melva_JEW_START_OPPOSE_TT
		add_mil_power = 25

		add_country_modifier = {
			name = melva_jews_value
			duration = -1
		}
		change_variable = {
			which = Melva_JEW_OPPOSE
			value = 10
		}
	}

	option = {
		name = Melva_Jew_Italy_Events.1.OPT2
		ai_chance = {
			factor = 50
		}

		custom_tooltip = Melva_JEW_START_APPROVE_TT
		add_dip_power = 25

		add_country_modifier = {
			name = melva_jews_value
			duration = -1
		}
		change_variable = {
			which = Melva_JEW_APPROVE
			value = 10
		}
	}
}

				value = 0
			}
			set_variable = {
				which = Melva_JEW_CITY_APPROVE
				value = 0
			}
		}
	}

	option = {
		name = Melva_Jew_Italy_Events.3.OPT1
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				owner = {
					check_variable = {
						which = Melva_JEW_OPPOSE
						value = 15
					}
				}
			}
			modifier = {
				factor = 2
				owner = { culture_group = iberian }
			}
		}

		owner = {
			add_prestige = 3
			add_papal_influence = 10
			add_mil_power = 10
			change_variable = {
				which = Melva_JEW_OPPOSE
				value = 5
			}
		}
		add_permanent_province_modifier = {
			name = italian_jew_province
			duration = -1
		}
		change_variable = {
			which = Melva_JEW_CITY_OPPOSE
			value = 10
		}
	}

	option = {
		name = Melva_Jew_Italy_Events.3.OPT2
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				owner = {
					check_variable = {
						which = Melva_JEW_APPROVE
						value = 7
					}
				}
			}
		}
		
		owner = { 
			add_legitimacy_equivalent = { amount = 3 republican_tradition = 3 }
			change_variable = {
				which = Melva_JEW_APPROVE
				value = 5
			}
		}
		if = {
			limit = { 
				owner = { ADM = 3 }
			}
			add_base_tax = 1
		}
		if = {
			limit = { 
				owner = { DIP = 3 }
			}
			add_base_production = 1
		}
		if = {
			limit = { 
				owner = { MIL = 3 }
			}
			add_base_manpower = 1
		}

		add_permanent_province_modifier = {
			name = italian_jew_province
			duration = -1
		}
		change_variable = {
			which = Melva_JEW_CITY_APPROVE
			value = 10
		}
		hidden_effect = {
			set_province_flag = melva_jew_sicily_giudecca_approved
		}
	}
}

#Expulsion of 1492
province_event = {
	id = Melva_Jew_Italy_Events.5
	title = Melva_Jew_Italy_Events.5.T
	desc = Melva_Jew_Italy_Events.5.D
	picture = RELIGIOUS_TURMOIL_eventPicture

	fire_only_once = yes
	trigger = {
		is_year = 1492
		NOT = { is_year = 1524 }
		OR = {
			area = western_sicily_area
			area = sicily_area
		}
		NOT = { province_id = 126 }
	}

	major = yes
	major_trigger = {
		italy_region = {
			owned_by = ROOT
		}
	}

	goto = ROOT

	mean_time_to_happen = {
		years = 1
	}

	option = {
		name = Melva_Jew_Italy_Events.5.OPT1
		ai_chance = {
			factor = 75
			modifier = {
				factor = 1.125
				owner = {
					check_variable = {
						which = Melva_JEW_OPPOSE
						value = 30
					}
				}
			}
			modifier = {
				factor = 1.25
				owner = { culture_group = iberian }
			}
			modifier = {
				factor = 1.125
				owner = { religious_ideas = 3 }
			}
		}

		owner = {
			add_papal_influence = 5
			add_mil_power = -10
		}
		remove_development_third_monarch_skill = yes
		if = {
			limit = {
				has_province_modifier = italian_jew_allowed_stay
			}
			remove_province_modifier = italian_jew_allowed_stay
			add_permanent_province_modifier = {
				name = italian_jew_province
			}
		}
		change_variable = {
			which = Melva_JEW_CITY_OPPOSE
			value = 15
		}
		random_list = {
			33 = {
				123 = { 
					random_list = {
						50 = { add_development_third_monarch_skill = yes }
						50 = { add_prosperity = 5 }
					}
				}
			}
			33 = {
				2984 = { 
					random_list = {
						50 = { add_development_third_monarch_skill = yes }
						50 = { add_prosperity = 5 }
					}
				}
			}
			33 = {
				1934 = { 
					random_list = {
						50 = { add_development_third_monarch_skill = yes }
						50 = { add_prosperity = 5 }
					}
				}
			}
		}
	}

	option = {
		name = Melva_Jew_Italy_Events.5.OPT2
		ai_chance = {
			factor = 25
			modifier = {
				factor = 4
				owner = {
					check_variable = {
						which = Melva_JEW_APPROVE
						value = 30
					}
				}
			}
			modifier = {
				factor = 0.75
				owner = { religious_ideas = 3 }
			}
			modifier = {
				factor = 2.5
				owner = { humanist_ideas = 3 }
			}
		}

		owner = {
			add_dip_power = 10
		}
		if = {
			limit = {
				has_province_modifier = italian_jew_allowed_stay
			}
			remove_province_modifier = italian_jew_allowed_stay
			add_permanent_province_modifier = {
				name = italian_jew_allowed_stay_2
				duration = -1
			}
		}
		else = {
			add_permanent_province_modifier = {
				name = italian_jew_allowed_stay
				duration = -1
			}
		}
		change_variable = {
			which = Melva_JEW_CITY_APPROVE
			value = 15
		}
	}
}
