country_event = { 
	id = ME_Westphalian_Events.7 #serfdom
	title = ME_Westphalian_Events.7.t 
	desc = ME_Westphalian_Events.7.d
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	option = { 
		name = ME_Westphalian_Events.7.a
		
		ai_chance = { factor = 1 }
	
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = 5
		}
		custom_tooltip = WES_transfer_20_percent
		hidden_effect = {
			random_owned_province = {
				limit = {
					development = 15
					NOT = { has_manufactory_trigger = yes }
				}
				build_possible_manufactory_in_province = yes
			}
			hidden_effect = {
				set_variable = {
					which = RHN_dev
					value = 0
				}
				every_owned_province = {
					limit = {
						NOT = { development = 15 }
					}
					export_to_variable = {
						which = RHN_dev
						value = trigger_value:development
					}
					PREV = {
						change_variable = {
							which = RHN_dev
							which = PREV
						}
					}
					set_variable = {
						which = RHN_dev
						value = 0
					}
				}
				multiply_variable = {
					which = RHN_dev
					value = 0.2
				}
				while = {
					limit = {
						check_variable = {
							which = RHN_dev
							value = 1
						}
					}
					random_owned_province = {
						limit = {
							NOT = { development = 15 }
							OR = {
								base_tax = 2
								base_production = 2
								base_manpower = 2
							}
						}
						if = {
							limit = {
								base_production = 2
								base_manpower = 2
								base_tax = 2
							}
							random_list = {
								33 = { add_base_tax = -1 }
								33 = { add_base_production = -1 }
								33 = { add_base_manpower = -1 }
							}
						}
						else_if = {
							limit = {
								base_production = 2
								base_manpower = 2
								NOT = { base_tax = 2 }
							}
							random_list = {
								50 = { add_base_production = -1 }
								50 = { add_base_manpower = -1 }
							}
						}
						else_if = {
							limit = {
								base_production = 2
								NOT = { base_manpower = 2 }
								NOT = { base_tax = 2 }
							}
							add_base_production = -1
						}
						else_if = {
							limit = {
								base_production = 2
								NOT = { base_manpower = 2 }
								base_tax = 2
							}
							random_list = {
								50 = { add_base_tax = -1 }
								50 = { add_base_manpower = -1 }
							}
						}
						else_if = {
							limit = {
								NOT = { base_production = 2 }
								NOT = { base_manpower = 2 }
								base_tax = 2
							}
							add_base_tax = -1
						}
						else_if = {
							limit = {
								base_production = 2
								NOT = { base_manpower = 2 }
								base_tax = 2
							}
							random_list = {
								50 = { add_base_tax = -1 }
								50 = { add_base_manpower = -1 }
							}
						}
						else_if = {
							limit = {
								NOT = { base_production = 2 }
								base_manpower = 2
								NOT = { base_tax = 2 }
							}
							add_base_manpower = -1
						}
					}
					subtract_variable = {
						which = RHN_dev
						value = 1
					}
					random_owned_province = {
						limit = {
							development = 15
						}
						random_list = {
							33 = { add_base_tax = 1 }
							33 = { add_base_production = 1 }
							33 = { add_base_manpower = 1 }
						}
					}
				}
				set_variable = {
					which = RHN_dev
					value = 0
				}
			}
		}
	}
	
	option = {
		name = ME_Westphalian_Events.7.b
		
		ai_chance = { factor = 1 }

		hidden_effect = {
			while = {
				limit = {

						estate = estate_nobles
