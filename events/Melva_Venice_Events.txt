				tag = ITA
			}
		}
		is_year = 1453
		owns_core_province = 112
		navy_size_percentage = 0.75
	}

	option = {
		name = Melva_Venice_Events.3.OPT1
		ai_chance = {
			factor = 50
		}

		add_country_modifier = {
			name = ven_naval_reforms
			duration = 5475
		}
		if = {
			limit = {
				has_dlc = "Golden Century"
				has_flagship = no
				capital_scope = {
					has_port = yes
				}
			}
			capital_scope = {
				create_flagship = {
					name = "Leone Alato"
					type = galley
				}
			}
		}
		else_if = {
			limit = {
				any_owned_province = {
					has_port = yes
				}
			}
			random_owned_province = {
				limit = {
					has_port = yes
				}
				create_flagship = {
					name = "Leone Alato"
					type = galley
				}
			}
		}
	}
}
