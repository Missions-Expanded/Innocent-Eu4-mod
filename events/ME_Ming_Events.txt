	option = {  # crush pirates
		name = ME_Ming_Events.14.a
		ai_chance = { factor = 50 }

		northern_kyushu = {
			limit = {
				NOT = { owned_by = MNG }

		kyushu_area = {
			limit = {
				NOT = { owned_by = MNG }

		}

		name = ME_Ming_Events.14.b

		add_mercantilism = 1
		add_corruption = -0.5
	}
}
