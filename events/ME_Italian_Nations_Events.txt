#Sottomission delle Repubbliche Marinare
country_event = {
	id = ME_Italian_Nations_Events.1
	title = ME_Italian_Nations_Events.1.T
	desc = ME_Italian_Nations_Events.1.D

	trigger = {
		OR = {
			AND = {
				tag = VEN
				is_in_war = {
					war_score = 75
					attacker_leader = ROOT
					defender_leader = GEN
				}
				GEN = {
					capital_scope = {
						sieged_by = ROOT
					}
				}
			}
			AND = {
				tag = GEN
				is_in_war = {
					war_score = 75
					attacker_leader = ROOT
					defender_leader = VEN
				}
				VEN = {
					capital_scope = {
						sieged_by = ROOT
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = ME_Italian_Nations_Events.1.OPT1
		ai_chance = {
			factor = 50
		}

	}

	option = {
		name = ME_Italian_Nations_Events.1.OPT2
		ai_chance = {
			factor = 50
		}

		if = {
			limit = {
				tag = VEN
			}
			GEN = {
				country_event = {
					id = ME_Italian_Nations_Events.2
				}
			}
		}
		else = {
			VEN = {
				country_event = {
					id = ME_Italian_Nations_Events.2
				}
			}
		}
	}
}

			change_trade_goods = glass

				limit = {

			}

		}
	}
