#Iron Mines
country_event = {
	id = ME_Ireland_Events.3
	title = ME_Ireland_Events.3.T
	desc = ME_Ireland_Events.3.D
	picture = ADVISOR_eventPicture

	is_triggered_only = yes

	goto = 4120

	option = {
		name = ME_Ireland_Events.3.OPT1

		ai_chance = {
			factor = 50
		}
		
		4120 = {
			if = {
				limit = {
					is_expanded_mod_active = { mod = trade_goods }
				}
				change_trade_goods = munitions
			}
			else = {
				change_trade_goods = iron
			}
			build_possible_manufactory_in_province_no_tech_requirement = yes
		}
	}

	option = {
		name = ME_Ireland_Events.3.OPT2

		ai_chance = {
			factor = 50
		}
		
		4120 = {
			add_base_production = 3
		}
	}
}
