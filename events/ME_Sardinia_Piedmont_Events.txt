				add_country_modifier = {
					name = lvn_union_crowns_subject
					duration = 18250
				}
			}
			add_country_modifier = {
				name = lvn_union_crowns_overlord
				duration = 18250
			}
		}
		else = {
			SAV = {
				add_country_modifier = {
					name = lvn_union_crowns_subject
					duration = 18250
				}
			}
